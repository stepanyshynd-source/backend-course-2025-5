# === GET: перший запит, картинка ще не в кеші, сервер завантажить з http.cat ===
curl http://127.0.0.1:3000/200 --output 200.jpg

# === GET: повторний запит — тепер картинка має прийти з кешу ===
curl http://127.0.0.1:3000/200 --output 200_cached.jpg

# === PUT: вручну покласти картинку у кеш (наприклад, 404.jpg) ===
curl -X PUT --data-binary "@200.jpg" http://127.0.0.1:3000/404

# === GET: перевірити, що PUT спрацював (повертає кешований файл) ===
curl http://127.0.0.1:3000/404 --output 404.jpg

# === DELETE: видалити файл із кешу ===
curl -X DELETE http://127.0.0.1:3000/404

# === GET після DELETE — файл зник, очікуємо 404 ===
curl -v http://127.0.0.1:3000/404

# === Перевірка методу, який не дозволено (наприклад, POST) ===
curl -X POST http://127.0.0.1:3000/200
